// GET /api/school-service/course-templates
async function getTemplates(params: {
  search?: string;
  courseType?: CourseType;
}) {
  return prisma.courseTemplate.findMany({
    where: {
      ...(params.search && {
        OR: [
          { templateName: { contains: params.search } },
          { description: { contains: params.search } },
        ],
      }),
      ...(params.courseType && { courseType: params.courseType }),
    },
    orderBy: [
      { isDefault: "desc" },
      { usageCount: "desc" },
      { createdAt: "desc" },
    ],
  });
}

// POST /api/school-service/course-templates/[id]/use
async function useTemplate(
  templateId: string,
  data: {
    schoolId: string;
    academicYear: string;
    startDate: Date;
    endDate?: Date;
    overrides?: Partial<CourseFormData>;
  }
) {
  const template = await prisma.courseTemplate.findUnique({
    where: { id: templateId },
  });

  if (!template) throw new Error("æ¨¡æ¿ä¸å­˜åœ¨");

  const course = await prisma.schoolCourse.create({
    data: {
      schoolId: data.schoolId,
      courseName: data.overrides?.courseName || template.courseName,
      courseType: template.courseType,
      courseTerm: template.courseTerm,
      academicYear: data.academicYear,
      startDate: data.startDate,
      endDate: data.endDate,

      maxStudents: template.maxStudents,
