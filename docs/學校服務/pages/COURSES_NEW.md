function calculateEstimatedRevenue(course: CoursePreview): number {
  switch (course.chargingModel) {
    case "STUDENT_PER_LESSON":
      // éœ€è¦é è¨ˆå­¸ç”Ÿäººæ•¸å’Œèª²å ‚æ•¸

        (course.maxStudents || 0) * (course.studentPerMonthFee || 0) * months

      return 0;

}
